**Step 7: Add the `generateReelCaption` helper function:**

Add this as a new async function BEFORE `handleIncomingMessage` (around line 620, near `processNewImageFlow`). CRITICAL: This function is defined at module scope, so it can access module-level imports (`savePost`, `randomUUID`, `getStylistName`, `composeFinalCaption`) but CANNOT access variables local to `handleIncomingMessage` (like `_stylistId`).
```javascript
/**
 * Generate a reel caption, create a draft, and send the preview to the stylist.
 * Reuses the same draft/preview/APPROVE flow as photo posts.
 *
 * NOTE: This is a module-level function. It can call getStylistName() (module-level)
 * and savePost() (imported). It CANNOT reference _stylistId (local to handleIncomingMessage).
 * The stylistId is derived from the stylist parameter instead.
 *
 * IMPORTANT: Calls composeFinalCaption() after generateCaption() to inject the
 * "Book: https://..." line. This matches the pattern in processNewImageFlow for photo
 * posts and ensures enqueuePost() UTM regex replacement can fire for reel posts.
 */
async function generateReelCaption({
  chatId, videoPublicUrl, serviceDescription,
  drafts, generateCaption, moderateAIOutput, sendMessage,
  stylist, salon,
}) {
  const salonInfo = salon?.salon_info || salon || {};
  const salonSlug = salonInfo.slug || salon?.salon_id || salon?.id || '';

  // Generate caption via GPT-4o with postType='reel'
  let caption;
  try {
    caption = await generateCaption({
      imageUrl: videoPublicUrl,
      notes: serviceDescription || '',
      salon: salonInfo,
      stylist,
      city: stylist?.city || '',
      postType: 'reel',
    });
  } catch (err) {
    console.error('[Router] Reel caption generation failed:', err.message);
    await sendMessage.sendText(chatId, "Sorry, couldn't generate a caption for your video. Please try again.");
    return;
  }

  // Moderate
  if (moderateAIOutput && typeof moderateAIOutput === 'function') {
    try {
      caption = await moderateAIOutput(caption) || caption;
    } catch {}
  }

  // Run through composeFinalCaption to inject "Book: https://..." line
  // This is the same pattern as processNewImageFlow for photo posts.
  // Without this, enqueuePost() UTM tracking injection (regex on "Book: https://...")
  // will never match for reel posts — breaking booking link tracking.
  const stylistId = stylist?.id || stylist?.stylist_id || null;
  const postId = randomUUID();
  try {
    caption = composeFinalCaption({
      baseCaption: caption,
      salon: salonInfo,
      stylist,
      postType: 'reel',
      salonId: salonSlug,
      postId,
    });
  } catch (err) {
    console.warn('[Router] composeFinalCaption failed for reel, using raw caption:', err.message);
    // Fall through with the un-composed caption — reel still works, just no Book: line
  }

  // Save draft to DB as post_type='reel'
  try {
    savePost({
      id: postId,
      salon_id: salonSlug,
      stylist_name: getStylistName(stylist),
      stylist_id: stylistId,
      image_url: videoPublicUrl,
      image_urls: JSON.stringify([videoPublicUrl]),
      base_caption: caption,
      final_caption: caption,
      post_type: 'reel',
      status: 'draft',
    });
  } catch (err) {
    console.warn('[Router] Could not persist reel draft to DB:', err.message);
  }

  // Store in-memory draft (same shape as photo drafts)
  const draft = {
    _db_id: postId,
    salon: salon,
    stylist,
    image_url: videoPublicUrl,
    image_urls: [videoPublicUrl],
    caption,
    postType: 'reel',
  };
  drafts.set(chatId, draft);

  // Send preview — "Here's your Reel caption:" (locked decision)
  const preview = `Here's your Reel caption:\n\n${caption}\n\nReply APPROVE to submit, EDIT <your changes> to modify, or REDO for a new caption.`;
  await sendMessage.sendText(chatId, preview);
}
```
