router.get("/day/:date", requireAuth, (req, res) => {
  const date = req.params.date;
  const salon_id = req.session.salon_id;

  // Validate YYYY-MM-DD format
  if (!/^\d{4}-\d{2}-\d{2}$/.test(date)) return res.status(400).send("Invalid date");

  const salon = db.prepare("SELECT timezone FROM salons WHERE slug = ?").get(salon_id);
  const tz = salon?.timezone || "America/Indiana/Indianapolis";

  // Compute UTC range for this local date
  const dayStart = DateTime.fromISO(date, { zone: tz }).startOf("day").toUTC().toFormat("yyyy-LL-dd HH:mm:ss");
  const dayEnd = DateTime.fromISO(date, { zone: tz }).endOf("day").toUTC().toFormat("yyyy-LL-dd HH:mm:ss");

  const posts = db.prepare(`
    SELECT p.id, p.post_type, p.status, p.scheduled_for, p.published_at, p.stylist_name,
           p.image_url, p.image_urls, p.final_caption, p.base_caption, p.vendor_campaign_id, vc.vendor_name
    FROM posts p
    LEFT JOIN vendor_campaigns vc ON p.vendor_campaign_id = vc.id
    WHERE p.salon_id = ?
      AND p.status NOT IN ('draft', 'cancelled')
      AND (p.scheduled_for BETWEEN ? AND ? OR p.published_at BETWEEN ? AND ?)
    ORDER BY COALESCE(p.scheduled_for, p.published_at) ASC
  `).all(salon_id, dayStart, dayEnd, dayStart, dayEnd);

  // Get CSRF token from req.csrfToken() for manual injection into fragment forms
  const csrfToken = typeof req.csrfToken === "function" ? req.csrfToken() : "";

  // Format the local date for display

  // Build HTML fragment (NOT wrapped in pageShell -- this is injected into the panel)
  let html = `

**2. POST `/reschedule` handler -- drag-to-reschedule:**

```javascript
router.post("/reschedule", requireAuth, (req, res) => {
  const salon_id = req.session.salon_id;
  const { postId, newDate } = req.body;

  // Validate inputs
  if (!postId || !/^\d{4}-\d{2}-\d{2}$/.test(newDate)) {
    return res.json({ ok: false, error: "Invalid input" });
  }

  // Verify ownership + status allows reschedule (only manager_approved posts with scheduled_for)
  const post = db.prepare(
    "SELECT scheduled_for FROM posts WHERE id = ? AND salon_id = ? AND status = 'manager_approved'"
  ).get(postId, salon_id);

  if (!post?.scheduled_for) {
    return res.json({ ok: false, error: "Post not found or not reschedulable" });
  }

  // Parse existing scheduled_for as UTC, replace only date portion, preserve time
  const original = DateTime.fromSQL(post.scheduled_for, { zone: "utc" });
  const [y, mo, d] = newDate.split("-").map(Number);
  const updated = original.set({ year: y, month: mo, day: d });
  const newTimestamp = updated.toFormat("yyyy-LL-dd HH:mm:ss");

  db.prepare("UPDATE posts SET scheduled_for = ? WHERE id = ? AND salon_id = ?")
    .run(newTimestamp, postId, salon_id);

  res.json({ ok: true, newScheduledFor: newTimestamp });
});
```

**3. Update SortableJS onEnd handler in the calendar page's client-side JS:**

Replace the stub onEnd from Plan 01 with the full implementation:

```javascript
onEnd(evt) {
  if (evt.from === evt.to) return; // same day, no-op
  const postId = evt.item.dataset.id;
  const newDate = evt.to.dataset.date;
  if (!postId || !newDate) { evt.from.appendChild(evt.item); return; }

  const csrfToken = document.querySelector('meta[name="csrf-token"]')?.content || '';
  fetch('/manager/calendar/reschedule', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'X-CSRF-Token': csrfToken },
    body: JSON.stringify({ postId, newDate }),
  })
  .then(r => r.json())
  .then(data => {
    if (!data.ok) {
      evt.from.appendChild(evt.item); // revert on failure
      console.error('Reschedule failed:', data.error);
    }
  })
  .catch(() => evt.from.appendChild(evt.item));
}
```
