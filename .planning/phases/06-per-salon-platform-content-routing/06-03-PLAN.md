   ```js
   // POST /internal/vendors/reset-routing?secret=...&salon=<slug>
   // Resets a salon's platform_routing to NULL (use defaults)
   router.post('/reset-routing', requirePin, (req, res) => {
     const secret = req.query.secret || req.body.secret;
     if (secret !== process.env.INTERNAL_SECRET) return res.status(403).send('Forbidden');
     const salonSlug = req.query.salon || req.body.salon;
     if (!salonSlug) return res.status(400).send('salon param required');
     db.prepare(`UPDATE salons SET platform_routing = NULL WHERE slug = ?`).run(salonSlug);
     const redirectSecret = secret;
     res.redirect(`/internal/vendors?secret=${redirectSecret}&routing_reset=${salonSlug}`);
   });
   ```
